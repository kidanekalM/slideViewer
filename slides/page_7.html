// D3.js chart for market growth document.addEventListener('DOMContentLoaded', function() { const data = [ {segment: "SMEs Going Digital", current: 4.2, growth: 40, projection: 8.5}, {segment: "Freelance Workforce", current: 1.8, growth: 35, projection: 4.2}, {segment: "Mobile Money Users", current: 45, growth: 25, projection: 88}, {segment: "Construction Projects", current: 180, growth: 20, projection: 310} ]; const margin = {top: 20, right: 30, bottom: 50, left: 60}; const width = document.getElementById('chart').clientWidth - margin.left - margin.right; const height = document.getElementById('chart').clientHeight - margin.top - margin.bottom; const svg = d3.select("#chart") .append("svg") .attr("width", width + margin.left + margin.right) .attr("height", height + margin.top + margin.bottom) .append("g") .attr("transform", `translate(${margin.left},${margin.top})`); // X axis const x = d3.scaleBand() .domain(data.map(d => d.segment)) .range([0, width]) .padding(0.3); svg.append("g") .attr("transform", `translate(0,${height})`) .call(d3.axisBottom(x)) .selectAll("text") .style("text-anchor", "end") .attr("dx", "-.8em") .attr("dy", ".15em") .attr("transform", "rotate(-25)"); // Add Y axis const y = d3.scaleLinear() .domain([0, d3.max(data, d => d.projection) * 1.1]) .range([height, 0]); svg.append("g") .call(d3.axisLeft(y).tickFormat(d => d + (d >= 100 ? "B" : "M"))); // Create tooltip const tooltip = d3.select("#chart") .append("div") .style("opacity", 0) .attr("class", "tooltip"); // Add bars for current values svg.selectAll(".bar-current") .data(data) .join("rect") .attr("class", "bar") .attr("x", d => x(d.segment)) .attr("width", x.bandwidth() / 2) .attr("y", d => y(d.current)) .attr("height", d => height - y(d.current)) .attr("fill", (d, i) => { const colors = ["#00D4AA", "#F59E0B", "#3B82F6", "#EF4444"]; return colors[i]; }) .on("mouseover", function(event, d) { tooltip.transition() .duration(200) .style("opacity", .9); tooltip.html(`Current: ${d.current} ${d.current >= 100 ? "B" : "M"}`) .style("left", (event.pageX) + "px") .style("top", (event.pageY - 28) + "px"); }) .on("mouseout", function(d) { tooltip.transition() .duration(500) .style("opacity", 0); }); // Add bars for projected values svg.selectAll(".bar-projected") .data(data) .join("rect") .attr("class", "bar") .attr("x", d => x(d.segment) + x.bandwidth() / 2) .attr("width", x.bandwidth() / 2) .attr("y", d => y(d.projection)) .attr("height", d => height - y(d.projection)) .attr("fill", (d, i) => { const colors = ["#0A2540", "#B45309", "#1D4ED8", "#B91C1C"]; return colors[i]; }) .attr("opacity", 0.7) .on("mouseover", function(event, d) { tooltip.transition() .duration(200) .style("opacity", .9); tooltip.html(`Projected: ${d.projection} ${d.projection >= 100 ? "B" : "M"} (+${d.growth}%)`) .style("left", (event.pageX) + "px") .style("top", (event.pageY - 28) + "px"); }) .on("mouseout", function(d) { tooltip.transition() .duration(500) .style("opacity", 0); }); // Add legend const legend = svg.append("g") .attr("transform", `translate(${width - 150}, 0)`); legend.append("rect") .attr("width", 15) .attr("height", 15) .attr("fill", "#00D4AA"); legend.append("text") .attr("x", 20) .attr("y", 12) .text("Current (2024)") .style("font-size", "12px") .style("fill", "#64748B"); legend.append("rect") .attr("width", 15) .attr("height", 15) .attr("y", 20) .attr("fill", "#0A2540") .attr("opacity", 0.7); legend.append("text") .attr("x", 20) .attr("y", 32) .text("Projected (2027)") .style("font-size", "12px") .style("fill", "#64748B"); });
